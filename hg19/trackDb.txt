#Example Advanced Composite Track with subgroups
track subgroups
compositeTrack on
shortLabel Composite w/ subgroups
longLabel Example of a Composite Track with subgroups
subGroup1 variablenumbers Variable_Numbers variable1=Variable_1 variable2=Variable_2
subGroup2 variableletters Variable_Letters variableA=Variable_A variableB=Variable_B
dimensions dimX=variablenumbers dimY=variableletters
sortOrder variablenumbers=+ variableletters=+
type bigBed 3

    track bigBed1
    bigDataUrl data/bigBed/example_1_bed_3.bigBed
    shortLabel Var1 and VarA 
    longLabel Variable 1 and Variable A 
    type bigBed 3
    parent subgroups
    subGroups variablenumbers=variable1 variableletters=variableA

    track bigBed2
    bigDataUrl data/bigBed/example_2_bed_3.bigBed
    shortLabel Var2 and VarB
    longLabel Variable 2 and Variable B
    type bigBed 3
    parent subgroups
    subGroups variablenumbers=variable2 variableletters=variableB

    track bigBed3
    bigDataUrl data/bigBed/example_3_bed_3.bigBed
    shortLabel Var1 and VarB
    longLabel Variable 1 and Variable B
    type bigBed 3
    parent subgroups
    subGroups variablenumbers=variable1 variableletters=variableB
